title: [BOJ/Node.js] 2752 세수정렬
---

```javascript
const fs = require('fs');
let input = fs
    .readFileSync('./input.txt')
    .toString()
    .split(' ')
    .map((item) => +item);

solution = [];
for (let i = 0; i < 3; i++) {
    solution.push(input.splice(input.indexOf(Math.min(...input)), 1));
}

console.log(solution.join(' '));
```
