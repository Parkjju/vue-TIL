  startBtn.innerText = "Start Recording";

  startBtn.addEventListener("click", handleStart);

  startBtn.innerText = "Stop Recording";

    video: false,
  });
  audio.srcObject = stream;
  audio.play();

startBtn.addEventListener("click", handleStart);
