# 풀이

반전 문자열과 원본 문자열 하나씩 비교, 굳이 반으로 쪼개서 순회 X

import Foundation

let input = readLine()!
let reversed = String(input.reversed())

func solution() -> Int {
    var offset = 0

    for char in reversed {
        if char == input[input.index(input.startIndex, offsetBy: offset)] {
            offset += 1
            continue
        } else {
            return 0
        }
    }
    return 1
}

print(solution())
